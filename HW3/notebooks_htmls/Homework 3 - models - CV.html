Breast Cancer Coimbra Dataset¶

This work is a series of different classification models to predict Breast cancer. The goal is to predict whether the patient has or not Breast cancer.

This dataset was provided by [Patricio, 2018] Patrcio, M., Pereira, J., Crisstomo, J., Matafome, P., Gomes, M., Seia, R., & Caramelo, F. (2018). Using Resistin, glucose, age and BMI to predict the presence of breast cancer. BMC Cancer, 18(1) and it's available at https://archive.ics.uci.edu/ml/datasets/Breast+Cancer+Coimbra

def custom_describe(dataset):
    df = dataset.drop(columns=['Classification']).describe().T
    df = df.drop(columns=['count'])
    df['skewness'] = dataset.drop(columns=['Classification']).skew(axis=0).values
    df['maxmin_ratio'] = df['max']/df['min']
    return df

k = 10
print(f"\n# {k}-FOLD CV PERFORMANCE #")

kf = KFold(n_splits=k)

lr_kf_acc = []
i = 1
for train_index, test_index in kf.split(X):
    X_train, X_test = X.iloc[train_index], X.iloc[test_index]
    y_train, y_test = y[train_index], y[test_index]
    
    reg = LogisticRegression().fit(X_train, y_train)
    
    test_acc = round(accuracy_score(y_test, reg.predict(X_test)), 3)
    
    print(f"\n  FOLD-{i}")
    print(f"   Test Accuracy: {test_acc}")
    
    lr_kf_acc.append(test_acc)
    i+=1
    
print(f'''
Summary:
  {k}-fold Average Accuracy: {np.mean(lr_kf_acc)}
''')
